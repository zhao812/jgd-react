//初始配置流程
git config --global user.name "用户名"		//配置git 提交的用户名称
git config --global user.email "邮箱地址"	//配置git 提交的邮箱
git clone ssh://-----/****.git			//获取远程库


//功能开发流程
git pull 				//更新代码
git checkout -b devName 		//创建本地开发分支 devName 自定义名称
git add 文件				//添加所有新建文件
git commit -m "注释"			//提交文件并写注释
git checkout dev			//返回到开发分支
git merge --no-ff devName -m "注释"	//合并到dev分支并写注释
git push origin dev			//上传到服务器
git push origin :dev			//删除服务器上的分支

git branch -d devName			//删除本地分支
